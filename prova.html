function capitalrisk(cap, risk) {

    let caprisk = cap * risk;

    caprisk = caprisk / 100;

    return caprisk;

}


function positionsize(riskedcapital, entry, sl) {

    let position = riskedcapital / (entry - sl);

    // position = position.toFixed(3);

    return position;


}


function margin(positionsize, entry, leverage) {

    let margin = positionsize * entry;

    margin = margin / leverage;

    // margin = margin.toFixed(2);

    return margin;


}


function pnl(positionsize, exit, entry) {

    let pnl = exit - entry;

    pnl = positionsize * pnl;

    return pnl;


}


function riskreward(pnl, riskedcapital) {

    let riskreward = pnl / riskedcapital;

    return riskreward;


}


function roe(pnl, margin) {

    let roe = pnl / margin;

    roe = roe * 100;

    return roe;

}


function percentchange(entry, exit) {

    let percentchange = exit - entry;

    percentchange = percentchange / entry;

    percentchange = percentchange * 100;

    return percentchange;


}

function percentchangeforshort(entry, exit) {

    let percentchange = entry - exit;

    percentchange = percentchange / exit;

    percentchange = percentchange * 100;

    return percentchange;


}


function liquidation(entry, leverage) {

    let liquidation = entry / leverage;

    liquidation = entry - liquidation;

    liquidation = liquidation + (entry * 1.10) / 100;

    return liquidation;


}




function liquidationforshort(entry, leverage) {

    let liquidation = entry / leverage;

    liquidation = parseFloat(entry) + parseFloat(liquidation);

    let entryv = (entry * 1.10) / 100;

    liquidation = liquidation - entryv;

    return liquidation;


}
